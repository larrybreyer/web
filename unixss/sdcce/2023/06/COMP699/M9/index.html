Write a JavaScript function to filter false, null, 0 and blank values from
an array.

TestData: console.log(filterValues([58, '', 'abcd', true, null, false, 0]));

ExpectedResult: [58, "abcd", true]

i    function filterValues(oldList) {
        let newList = []; 
        for ( let data of oldList ) { 
            if(data != false && data != null && data != 0 && data != '') {
                newList.push(data);
            }   
        }   
        return newList;
    }
